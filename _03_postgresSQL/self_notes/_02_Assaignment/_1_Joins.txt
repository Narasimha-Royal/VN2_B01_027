id serial primary key,
fname varchar(30),
lname varchar(25),
age int,
salary int
);

create table department (
id serial primary key,
dept_name varchar(30),
dept_id int
)

select dept_id, fname, salary, dept_name from employe cross join department

select fname, dept_id from department cross join employe

select fname,salary,dept_name ,dept_id from employe inner join department
on employe.id=department.dept_id;

select fname,dept_name, dept_id from employe left outer join department 
on employe.id=department.dept_id

select fname,dept_name, dept_id from employe right outer join department 
on employe.id=department.dept_id

select fname,dept_name, dept_id from employe full outer join department 
on employe.id=department.dept_id
