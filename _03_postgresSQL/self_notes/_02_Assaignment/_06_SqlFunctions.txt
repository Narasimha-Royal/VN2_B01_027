create or replace function coursecost ()
returns integer as $totalcoursecost$
declare 
totalcoursecost integer ;
begin
	select sum(coursecost) into totalcoursecost from studycourses;
return totalcoursecost ;
end;
$totalcoursecost$ language plpgsql
select coursecost()
